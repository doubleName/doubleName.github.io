# vue-cli跨域设置

# vue.config.js的配置

module.exports = {
    publicPath: './',
    devServer: {
        port: 4000,
        proxy: {
            '/proxyApi': {
                target: 'http://localhost:8080',
                changeOrigin: true,
                pathRewrite: {
                    'proxyApi': ''
                }
            }
        }
    }
}

# main.js的配置

import Vue from 'vue'
import axios from 'axios'

axios.defaults.headers.post['Content-Type'] = 'application/x-www-form-urlencoded;charset=utf-8'
axios.defaults.baseURL = '/proxyApi'

Vue.prototype.$axios = axios
Vue.config.productionTip = false

new Vue({
  render: h => h(App)
}).$mount('#app')

# 使用文件中的配置

请求一条数据

export default {
    name: 'home1',
    data(){
        return {
            movieInfo: {},
        }
    }
    created:function(){
      this.getMovieInfo();
    },
    methods: {
        getMovieInfo: function (){
            this.$axios.get('/MovieHome/show/brief')
              .then((response)=>{             
                  this.movieInfo = response.data;
              })
              .catch(function(error) {
                  console.log(error)
              })
        }
    }
}
</script>

# 注意

  • 这里有个问题，console里会出现这样的提示。比如baseURL设置的/proxyApi，localhost是http://localhost:4000。 -请求失败时，即使设置了baseUrl,但是返回请求的baseUrl还是localhost。 -与设置无关，baseURL实际上是生效的，只是提示的错误是返回的url，跨域并没有问题。请求的仍是/proxyApi/MovieComment/Sh/messages/1,
GET http://localhost:4000/proxyApi/MovieComment/Sh/messages/1 404 (Not Found)
